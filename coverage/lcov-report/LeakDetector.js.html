  const array = [...arr];
  if (array.some(el => isNaN(el))) return false;
  const diff = Math.abs(array[1] - array[0]);
  return array.some((el, i, arr) => {
    if (i === arr.length - 1) return Math.abs(arr[i - 1] - el) !== diff;
    return Math.abs(arr[i + 1] - el) !== diff;
  });
}
